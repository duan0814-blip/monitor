async function loadEvents() { try { const response = await fetch('list.json?t=' + Date.now()); const json = await response.json(); // 關鍵修正：相容多種 API 回傳格式 const products = json.data?.products || json.products || []; const listContainer = document.getElementById('event-list'); listContainer.innerHTML = ''; if (products.length === 0) throw new Error('No data'); products.forEach(p => { // 自動判斷欄位名稱 (有些版本是 name，有些是 productName) const pName = p.productName || p.name || "未知活動"; const pSales = p.totalSales || p.sales || 0; const pUuid = p.productUuid || p.id; const card = document.createElement('div'); card.className = 'card p-6 rounded-xl border border-gray-800 shadow-xl'; card.innerHTML = `

${pName}

目前總銷量：${pSales}

`; listContainer.appendChild(card); }); } catch (e) { console.error(e); document.getElementById('event-list').innerHTML = '
數據解析失敗，請確認 list.json 內容。
'; } }
